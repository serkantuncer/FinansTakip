import os

a = Analysis(
    ['main.py'],
    pathex=[os.path.abspath('.')],  # Proje dizinini tanımladık
    binaries=[],
    datas=[
        ('templates', 'templates'), 
        ('static', 'static'), 
        ('instance/finans_takip.db', 'instance'),
        ('instance', 'instance')
    ],
    hiddenimports=[
        'sqlite3',
        'flask',
        'flask_login',
        'werkzeug',
    ],
    hookspath=[],
    hooksconfig={},
    runtime_hooks=[],
    excludes=[],
    noarchive=False,
    optimize=0,
)

pyz = PYZ(a.pure)

exe = EXE(
    pyz,
    a.scripts,
    a.binaries,
    a.datas,
    [],
    name='Financial Portal',
    debug=False,
    bootloader_ignore_signals=False,
    strip=False,
    upx=True, # UPX bazen virüs tarayıcıları tarafından yanlış pozitif verebilir. Gerekirse False yap

    console=False,  # İstersen hata ayıklama için True yap
    disable_windowed_traceback=False,
    argv_emulation=False,
    target_arch=None,
    codesign_identity=None,
    entitlements_file=None,
